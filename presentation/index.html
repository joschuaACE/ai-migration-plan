Before & After

CBaKAlternativ — same business semantics, modern idioms

C++ — Alternativ.cpp

// Global mutable state
CBaKAlternativ KeineUmrechnung(
    CBaKAlternativ::eBasis, 0, 0);

CBaKAlternativ::CBaKAlternativ(
    const EAusgleich e,
    const CBaKConversionData* const pCD,
    CBaKSsWaehrungsUmrechnung* pWUm) :
  m_eStatus(e), m_pConversionData(pCD),
  m_pWUm(pWUm) {
    BIASSERT((!m_pConversionData
      && m_eStatus==eBasis && !m_pWUm)
      || (m_pConversionData
      && m_eStatus!=eBasis && m_pWUm));
}

// operator bool() overload
CBaKAlternativ::operator bool() const {
    return m_eStatus != eBasis;
}

CBaKCurrency CBaKAlternativ::RechneUm(
    const CBaKCurrency& cur) const {
    if (m_eStatus == eBasis) {
        BIASSERT(false);
        return cur;
    }
    CBaKCurrency result;
    m_pWUm->RechneUm(cur, result,
        m_pConversionData);
    return result;
}

Java 25 — Alternativ.java

// Immutable, null-safe, comparable
public final class Alternativ
    implements Comparable<Alternativ> {

  public static final Alternativ
    KEINE_UMRECHNUNG =
      new Alternativ(EAusgleich.BASIS,
          null, null);

  private final EAusgleich status;
  private final ConversionData data;
  private final SsWaehrungsUmrechnung umr;

  // Explicit method replaces operator bool()
  public boolean isActive() {
      return status != EAusgleich.BASIS;
  }

  // Safe delegation, no assertions needed
  public BigDecimal rechneUm(
      BigDecimal basisWert) {
    if (!isActive()
        || data == null || umr == null) {
      return basisWert;
    }
    return umr.rechneUm(basisWert, data);
  }

  // equals, hashCode, compareTo
  // with proper value semantics
}
